```c++
class Solution {
public:
    int reverse(int x) {
        if(x == INT_MIN)
            return 0;
        int num = abs(x);
        int result = 0;
        int preResult = 0;
        while(num!=0){
            result = result*10 + num%10;
            if(result/10!=preResult)
                return 0;
            preResult = result;
            num/=10;
        }
        if(x>0)
            return result;
        else
            return -result;
    }
};
```
